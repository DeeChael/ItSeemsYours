import List, ArrayList in isy.Util;

class Example {
    static void main(Array<Str> args) {
        List<Str> list = new ArrayList<>();
        list.add("a");
        list.add("b");
        list.add("c");
        list.add("d");
        System.print("Elements: ");
        foreach Str str in list as Loop loop {
            System.print(str);
            if (loop.hasNext()) {
                System.print(", ");
            }
        }
        System.print("\n");
        System.println("Hello, Isy!");
    }
}
